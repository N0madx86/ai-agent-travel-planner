    calangute_hotels = [

    ]

This is genuinely clever because:

✅ "Goa, Calangute" never triggers a scrape if "Goa" is cached
✅ One broad scrape covers all neighborhood variants
✅ Zero extra DB storage
✅ Instant response for subset searches

The logic in code is literally:
pythonasync def search_hotels(self, destination, checkin, checkout, budget):
    city = normalize_destination(destination)        # "goa"
    neighborhood = extract_neighborhood(destination) # "calangute" or None

    # Step 1 — try broad cache first
    hotels = cache.get(f"hotels:{city}")

    # Step 2 — broad cache miss → wide scrape
    if not hotels:
        hotels = await scraper.search(city, max_pages=5)
        cache.set(f"hotels:{city}", hotels, ttl=86400)

    # Step 3 — filter by neighborhood if specified
    if neighborhood:
        filtered = [
            h for h in hotels
            if neighborhood in h["location"].lower()
        ]
        # Step 4 — not enough? specific scrape
        if len(filtered) < 20:
            specific = await scraper.search(destination, max_pages=3)
            # merge without duplicates
            existing_names = {h["name"] for h in filtered}
            filtered += [h for h in specific if h["name"] not in existing_names]
            cache.set(f"hotels:{city}:{neighborhood}", filtered, ttl=86400)
        return filtered

    return hotels
The beauty of this approach:

✅ Wide scrape happens once, covers most neighborhood queries for free
✅ Specific scrape only fires when genuinely needed
✅ Results get cached at neighborhood level too so second search is instant
✅ Scales naturally — popular cities get wide cached, obscure ones get specific
