pythondef get_cache(city):
    path = f"cache/{city}.json"
    if os.path.exists(path):
        with open(path) as f:
            return json.load(f)
    return {}

def save_cache(city, key, hotels):
    data = get_cache(city)
    data[key] = {
        "hotels": hotels,
        "cached_at": datetime.now().isoformat()
    }
    with open(f"cache/{city}.json", "w") as f:
        json.dump(data, f)
